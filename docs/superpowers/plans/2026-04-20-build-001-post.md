## Task 1: Process the Dougal source photograph

- [ ] **Step 1: Write the one-off processing script**

Create `scripts/process-dougal.mjs`:

```javascript
// One-off script — processes the private Dougal source JPEG into
// public/dougal-source.jpg per spec §4. Delete this file after running.
import sharp from 'sharp';
import { mkdir } from 'node:fs/promises';
import path from 'node:path';

const SRC = '/Users/jodre11/Library/CloudStorage/OneDrive-Personal/Documents/Dougal Debug Limited/Branding/Dougal in glasses.JPEG';
const OUT = path.resolve('public/dougal-source.jpg');

async function main() {
    await mkdir(path.dirname(OUT), { recursive: true });

    const input = sharp(SRC).rotate(); // honour EXIF orientation before we strip metadata
    const meta = await input.metadata();

    // Tight crop to head and shoulders: centred horizontally, upper portion vertically,
    // producing a roughly 3:4 portrait. Adjust offsets after visual review if needed.
    const cropW = Math.round(meta.width * 0.55);
    const cropH = Math.round(cropW * 4 / 3);
    const left = Math.round((meta.width - cropW) / 2);
    const top = Math.round(meta.height * 0.08);

    await input
        .extract({ left, top, width: cropW, height: Math.min(cropH, meta.height - top) })
        .resize({ width: 1200, withoutEnlargement: true })
        .jpeg({ quality: 82, mozjpeg: true })
        // Deliberately no .withMetadata() — default strips EXIF (location, camera, timestamp).
        .toFile(OUT);

    console.log(`Wrote ${OUT}`);
}

main().catch((err) => {
    console.error(err);
    process.exit(1);
});
```
